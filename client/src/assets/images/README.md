```vue
<template>
  <!-- ж–№ејҸ1: дҪҝз”ЁGameImageз»„д»¶ -->

  <!-- ж–№ејҸ2: зӣҙжҺҘеј•е…Ҙ -->
  <img :src="Resources.ui.iconHp" />
</template>

<script>
import Resources from '@/assets/resources'
import { getPlayerImage, getEquipmentIcon } from '@/utils/resourceHelper'

export default {
  computed: {
    playerImg() {
      return getPlayerImage('male', 3)  // зӯ‘еҹәжңҹз”·жҖ§
    },
    swordImg() {
      return getEquipmentIcon('sword', 'epic')  // еҸІиҜ—еү‘
    }
  }
}
</script>
```
